181             // http://www.w3.org/Protocols/rfc2109/rfc2109
182             // Syntax:  General
183             // The two state management headers, Set-Cookie and Cookie, have common
184             // syntactic properties involving attribute-value pairs.  The following
185             // grammar uses the notation, and tokens DIGIT (decimal digits) and
186             // token (informally, a sequence of non-special, non-white space
187             // characters) from the HTTP/1.1 specification [RFC 2068] to describe
188             // their syntax.
189             // av-pairs   = av-pair *(";" av-pair)
190             // av-pair    = attr ["=" value] ; optional value
191             // attr       = token
192             // value      = word
193             // word       = token | quoted-string
194             
195             // http://www.ietf.org/rfc/rfc2068.txt
196             // token      = 1*<any CHAR except CTLs or tspecials>
197             // CHAR       = <any US-ASCII character (octets 0 - 127)>
198             // CTL        = <any US-ASCII control character
199             //              (octets 0 - 31) and DEL (127)>
200             // tspecials  = "(" | ")" | "<" | ">" | "@"
201             //              | "," | ";" | ":" | "\" | <">
202             //              | "/" | "[" | "]" | "?" | "="
203             //              | "{" | "}" | SP | HT

207             return (new RegExp("^[^\\x00-\\x20\\x7f\\(\\)<>@,;:\\\\\\\"\\[\\]\\?=\\{\\}\\/\\u0080-\\uffff]+\x24")).test(key);

402             	me._point = e.point;

405             	me.show();

1122                 } else {
1123                     if (startStatus == BMAP_ROUTE_STATUS_EMPTY) {
1124                         tip = "找不到相关的起点";

1126                     if (endStatus == BMAP_ROUTE_STATUS_EMPTY) {
